@if(!string.IsNullOrEmpty(statusMessage))

@code {
    private List<UserViewModel> users = new();
    private bool isLoading = true;
    string statusMessage = "";
    
    protected override async Task OnInitializedAsync()
    {
        var authUser = await JS.InvokeAsync<AuthUser>("firebaseAuth.getUser");
        if (authUser == null || !authUser.Email.Equals("admin@lumix.edu", StringComparison.OrdinalIgnoreCase))
        {
            Nav.NavigateTo("/Login");
            return;
        }
        await LoadData();
    }

    private async Task LoadData()
    {
        isLoading = true;
        try
        {
            users = await JS.InvokeAsync<List<UserViewModel>>("firebaseInterop.getAllUsers");
        }
        catch (Exception ex)
        {
            statusMessage = "Error connecting to database: " + ex.Message;
        }
        finally
        {
            isLoading = false;
        }
    }

    public class UserViewModel { public string Id { get; set; } public string Email { get; set; } public string Role { get; set; } public string Created { get; set; } }
    public class AuthUser { public string Email { get; set; } }

    private async Task ConfirmDelete(string id)
    {
        if(await JS.InvokeAsync<bool>("confirm", "Are you sure you want to remove this user from the system?"))
        {
            statusMessage = "";
            try
            {
                await JS.InvokeVoidAsync("firebaseInterop.deleteUser", id);
                await LoadData();
                statusMessage = "User successfully removed.";
            }
            catch (Exception ex)
            {
                statusMessage = "Error deleting user: " + ex.Message;
            }
        }
    }
}
